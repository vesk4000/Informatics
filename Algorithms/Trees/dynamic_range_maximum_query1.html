 23  const int TREE_MAX = ceil(log2(MAXA));
 24  int tree[1 << TREE_MAX];

 27  void tree_update(int index, int value)
 28  {
 29      index = tree_get_leaf(index);
 30      tree[index] = value;
 31      index = tree_get_parent(index);
 32      while(index >= 0)
 33      {
 34          tree[index] = max(tree[tree_get_left_child(index)], tree[tree_get_right_child(index)]);
 35          index = tree_get_parent(index);
 36      }
 37  }
 38  
 39  int tree_query(int left, int right)
 40  {
 41      left = tree_get_leaf(left);
 42      right = tree_get_leaf(right);
 43      int ans = max(tree[left], tree[right]);
 44      while(left + 1 < right)
 45      {
 46          if(is_left(left))
 47          {
 48              ans = max(ans, tree[left + 1]);
 49          }
 50          if(is_right(right))
 51          {
 52              ans = max(ans, tree[right - 1]);
 53          }
 54          left = tree_get_parent(left);
 55          right = tree_get_parent(right);
 56      }
 57      return ans;
 58  }

 60  void tree_build()
 61  {
 62      tree_height = ceil(log2(n)) + 1;
 63      for(int i = 0; i < n; ++i)
 64      {
 65          tree[tree_get_leaf(i)] = a[i];
 66      }
 67      for(int i = tree_get_leaf(0) - 1; i >= 0; --i)
 68      {
 69          tree[i] = max(tree[tree_get_left_child(i)], tree[tree_get_right_child(i)]);
 70      }
 71      tree_print();
 72  }

 74  void tree_print()
 75  {
 76      int lvl = 1;
 77      int lvl_curr = 0;
 78      for(int i = 0; i <= tree_get_leaf(n - 1); ++i)
 79      {
 80          cout << tree[i] << " ";
 81          lvl_curr++;
 82          if(lvl_curr >= lvl)
 83          {
 84              lvl_curr = 0;
 85              cout << endl;
 86              lvl *= 2;
 87          }
 88      }
 89  }

 92  {
 93      return x + pow2(tree_height - 1) - 1;

111  int tree_get_parent(int x)
112  {
113      if(is_left(x))
114      {
115          x += 1;
116      }
117      x /= 2;
118      x--;
119      return x;
120  }

122  bool is_left(int x)
123  {
124      return x % 2 == 1;
125  }

127  bool is_right(int x)
128  {
129      return !is_left(x);
